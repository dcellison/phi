\section{Examples}
Following are several examples of Phi phrases in formal mode. They are given in romanized text and English, followed by the Tengwar script.
\setlength{\parskip}{1.5em}

{\parindent0pt
\emph{si phiathe soa mia na thanie te phi -- My name is Daniel.}
\setlength{\parskip}{0.3em}

\ttmedium ⸱⸱⸱⸱⸱⸱⸱ \normalfont
\setlength{\parskip}{1.5em}

\emph{si mia na thi te lothea -- I love you.}
\setlength{\parskip}{0.3em}

\ttmedium ⸱⸱⸱⸱⸱ \normalfont
\setlength{\parskip}{1.5em}

\emph{si whetho na mipho te phi -- The sky is blue.}
\setlength{\parskip}{0.3em}

\ttmedium ⸱⸱⸱⸱⸱ \normalfont
\setlength{\parskip}{1.5em}

\emph{si lo sha na noshea te lephe -- They like food.}
\setlength{\parskip}{0.3em}

\ttmedium ⸱⸱⸱⸱⸱⸱ \normalfont
\setlength{\parskip}{1.5em}

\emph{si mia na sha te lephe -- I like it.}
\setlength{\parskip}{0.3em}

\ttmedium ⸱⸱⸱⸱⸱ \normalfont
\setlength{\parskip}{1.5em}

\emph{ho si mia na hashe te me lephe -- I do not like green!}
\setlength{\parskip}{0.3em}

\ttmedium ⸱⸱⸱⸱⸱⸱⸱ \normalfont
\setlength{\parskip}{1.5em}

\emph{si mia na miwho soa mia te li lephe -- I liked my friend.}
\setlength{\parskip}{0.3em}

\ttmedium ⸱⸱⸱⸱⸱⸱⸱⸱ \normalfont
\setlength{\parskip}{1.5em}

\emph{si noshea na soa lo mia te phi -- The food is ours.}
\setlength{\parskip}{0.3em}

\ttmedium ⸱⸱⸱⸱⸱⸱⸱ \normalfont

\emph{si noshea soa lo mia na shishoa te phi -- Our food is good.}
\setlength{\parskip}{0.3em}

\ttmedium ⸱⸱⸱⸱⸱⸱⸱⸱ \normalfont
\setlength{\parskip}{1.5em}

\emph{wha si thi na ropho shishoa te la phe -- Are you in good health?}
\setlength{\parskip}{0.3em}

\ttmedium ⸱⸱⸱⸱⸱⸱⸱⸱ \normalfont
\setlength{\parskip}{1.5em}

\emph{si lo mia na lo lipha sathe te lothea -- We love the orange trees.}
\setlength{\parskip}{0.3em}
